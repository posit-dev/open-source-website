{{- /* Related posts section for blog single pages Shows 3 posts from the same topic */ -}} {{- $related := slice -}} {{- $primaryTopic := "" -}} {{- /* Get primary topic (first in list) */ -}} {{- with .Params.topics -}} {{- $primaryTopic = index . 0 -}} {{- end -}} {{- if $primaryTopic -}} {{- /* Find posts in same topic, exclude current post */ -}} {{- $site := .Site -}} {{- range $site.Taxonomies.topics -}} {{- $termKey := .Page.Title -}} {{- if eq $termKey $primaryTopic -}} {{- range .Pages -}} {{- if ne .RelPermalink $.RelPermalink -}} {{- $related = $related | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Fallback: if < 3 related posts, add recent posts */ -}} {{- if lt (len $related) 3 -}} {{- $needed := sub 3 (len $related) -}} {{- $recentPosts := where .Site.RegularPages "Section" "blog" | first (add (len $related) $needed 1) -}} {{- range $recentPosts -}} {{- if and (ne .RelPermalink $.RelPermalink) (lt (len $related) 3) -}} {{- $found := false -}} {{- range $related -}} {{- if eq .RelPermalink $.RelPermalink -}} {{- $found = true -}} {{- end -}} {{- end -}} {{- if not $found -}} {{- $related = $related | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Only show section if we have related posts */ -}} {{- if gt (len $related) 0 -}}

{{- if $primaryTopic -}} More On {{ $primaryTopic | title }} {{- else -}} Related Posts {{- end -}}

{{- range first 3 $related -}} {{ partial "item.html" (dict "page" . "hide-badge" true) }} {{- end -}}
{{- end -}}
