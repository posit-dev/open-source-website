{{/* event-section.html — Classify an event as "current", "upcoming", or "past". Input: a Page object with .Params.start_date and optionally .Params.end_date Returns: string "current", "upcoming", or "past" */}} {{- $page := . -}} {{- $now := now -}} {{- $section := "past" -}} {{- $startDate := $page.Params.start_date -}} {{- $endDate := $page.Params.end_date -}} {{- if and $startDate $endDate -}} {{- $start := time $startDate -}} {{- $end := time $endDate -}} {{- if and (le $start $now) (ge $end $now) -}} {{- $section = "current" -}} {{- else if gt $start $now -}} {{- $section = "upcoming" -}} {{- end -}} {{- else if $startDate -}} {{- $start := time $startDate -}} {{- if gt $start $now -}} {{- $section = "upcoming" -}} {{- end -}} {{- end -}} {{- return $section -}}
