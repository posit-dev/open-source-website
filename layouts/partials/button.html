{{/* button: Render a styled link as a button. Required: url Optional: text, target, icon (alias for icon-left), icon-left, icon-right, alt (aria-label), class, size ("small", "medium", "large") External URLs (http/https) automatically get target="_blank" and a tabler--external-link icon on the right. Set icon-right="false" to suppress the automatic external icon. */}} {{ $url := .url }} {{ $text := .text | default "" }} {{ $target := .target | default "" }} {{ $iconLeft := index . "icon-left" | default (index . "icon" | default "") }} {{ $iconRight := index . "icon-right" | default "" }} {{ $alt := .alt | default "" }} {{ $class := .class | default "" }} {{ $size := .size | default "medium" }} {{ $isExternal := or (hasPrefix $url "http://") (hasPrefix $url "https://") }} {{/* Auto-set target="_blank" for external URLs unless explicitly overridden */}} {{ if and $isExternal (eq $target "") }} {{ $target = "_blank" }} {{ end }} {{/* Auto-set external icon on the right unless suppressed */}} {{ if and $isExternal (ne $iconRight "false") (eq $iconRight "") }} {{ $iconRight = "tabler--external-link" }} {{ end }} {{ if eq $iconRight "false" }} {{ $iconRight = "" }} {{ end }} {{ $sizeClass := "px-4 py-3 text-sm leading-5 gap-2" }} {{ $iconSizeLeft := "w-5 h-5" }} {{ $iconSizeRight := "w-4 h-4" }} {{ if eq $size "small" }} {{ $sizeClass = "px-3 py-1.5 text-xs leading-4 gap-1.5" }} {{ $iconSizeLeft = "w-4 h-4" }} {{ $iconSizeRight = "w-3 h-3" }} {{ else if eq $size "large" }} {{ $sizeClass = "px-6 py-3 text-base leading-6 gap-2" }} {{ $iconSizeLeft = "w-5 h-5" }} {{ $iconSizeRight = "w-5 h-5" }} {{ end }} {{ $baseClass := printf "btn-shortcode inline-flex mb-5 mr-5 items-center %s rounded-lg bg-blue-500 text-white font-semibold align-middle hover:bg-blue-600 transition no-underline" $sizeClass }} {{ $finalClass := trim (printf "%s %s" $baseClass $class) " " }} {{- with $iconLeft }}{{ end -}} {{- with $text }}{{ . }}{{ end -}} {{- with $iconRight }}{{ end -}}
